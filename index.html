API Architecture — How to Wire It Up

Never call paid APIs directly from the browser. It exposes your API keys and has no caching. The correct pattern is a serverless proxy:

Browser → fetch('/api/news') → Vercel Serverless Function → NewsAPI
                                    ↕ caches for 5 min
                               no key exposure, no CORS issues

Create /api/news.js on Vercel:

// api/news.js — Vercel serverless function
export default async function handler(req, res) {
  // Simple in-memory cache (5 min)
  const cache = global._newsCache;
  if (cache && Date.now() - cache.ts < 300000) {
    return res.json(cache.data);
  }

  const r = await fetch(
    `https://newsapi.org/v2/top-headlines?language=en&pageSize=20`,
    { headers: { 'X-Api-Key': process.env.NEWS_API_KEY } }
  );
  const data = await r.json();

  global._newsCache = { data, ts: Date.now() };
  res.setHeader('Cache-Control', 's-maxage=300');
  res.json(data);
}

Set your API key as an environment variable in Vercel dashboard — never hardcode it in the file. Then in your HTML:

async function loadLiveNews() {
  const res = await fetch('/api/news');
  const { articles } = await res.json();
  const mapped = articles.map(a => ({
    hl: a.title,

    url: a.url,
    mins: Math.floor((Date.now() - new Date(a.publishedAt)) / 60000),
    // ... map to your NEWS schema

  renderNews('breakingGrid', mapped.slice(0, 6));
}

Why You Need a Backend

Right now, votes, bookmarks, and comments live in localStorage — they vanish when the user clears their browser, don't sync across devices, and can't be shared. A real backend gives you:

1
Persistent user data — bookmarks and vote history survive browser clears and work on any device
2
Real community — comments visible to all users, not just the person who posted them
3
Vote integrity — prevent users from voting 100 times; tie votes to accounts
4
Subscriber list ownership — email addresses you own, not trapped in a third-party tool
5
Analytics — understand which stories, tools, and tabs actually get engagement
Recommended Stack: Supabase

Supabase is a hosted Postgres database with a built-in REST API, real-time subscriptions, file storage, and OAuth authentication — all accessible from the browser via a JavaScript client. No server code needed for basic operations.

Database Schema

Run this SQL in the Supabase dashboard to create your tables:

-- Users (auto-created by Supabase Auth, just add profile data)
CREATE TABLE profiles (
  id UUID REFERENCES auth.users PRIMARY KEY,
  username TEXT UNIQUE,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Comments
CREATE TABLE comments (
  id BIGSERIAL PRIMARY KEY,
  user_id UUID REFERENCES profiles(id),
  text TEXT NOT NULL CHECK (length(text) BETWEEN 1 AND 1000),
  likes INT DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Votes (one per user per story)
CREATE TABLE votes (
  user_id UUID REFERENCES profiles(id),
  story_id INT NOT NULL,
  direction TEXT CHECK (direction IN ('up','down')),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (user_id, story_id)
);

-- Bookmarks
CREATE TABLE bookmarks (
  user_id UUID REFERENCES profiles(id),
  story_id INT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (user_id, story_id)
);

-- Subscribers
CREATE TABLE subscribers (
  id BIGSERIAL PRIMARY KEY,
  email TEXT UNIQUE NOT NULL,
  subscribed_at TIMESTAMPTZ DEFAULT NOW(),
  active BOOLEAN DEFAULT TRUE
);

-- Tool submissions
CREATE TABLE tool_submissions (
  id BIGSERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  url TEXT NOT NULL,
  category TEXT NOT NULL,
  pricing TEXT NOT NULL,
  description TEXT NOT NULL,
  contact_email TEXT NOT NULL,
  plan TEXT CHECK (plan IN ('standard','featured')),
  stripe_session_id TEXT,
  status TEXT DEFAULT 'pending',
  submitted_at TIMESTAMPTZ DEFAULT NOW()
);

Then add Row Level Security so users can only touch their own rows:

-- Enable RLS on all tables
ALTER TABLE comments ENABLE ROW LEVEL SECURITY;
ALTER TABLE votes ENABLE ROW LEVEL SECURITY;
ALTER TABLE bookmarks ENABLE ROW LEVEL SECURITY;

-- Comments: anyone can read, only author can delete their own
CREATE POLICY "Anyone can read comments" ON comments FOR SELECT USING (TRUE);
CREATE POLICY "Users insert own comments" ON comments FOR INSERT
  WITH CHECK (auth.uid() = user_id);
CREATE POLICY "Users delete own comments" ON comments FOR DELETE
  USING (auth.uid() = user_id);

-- Votes: users can only read/write their own
CREATE POLICY "Users manage own votes" ON votes
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

-- Bookmarks: users can only read/write their own
CREATE POLICY "Users manage own bookmarks" ON bookmarks
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

// OLD (localStorage)
function toggleBk(e, id) {

// NEW (Supabase)
async function toggleBk(e, id) {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) { openLoginModal(); return; }

  const { data: existing } = await supabase
    .from('bookmarks')
    .select('*')
    .eq('user_id', user.id)
    .eq('story_id', id)
    .single();

  if (existing) {
    await supabase.from('bookmarks').delete()
      .eq('user_id', user.id).eq('story_id', id);
    showToast('Removed from saved');
  } else {
    await supabase.from('bookmarks').insert({ user_id: user.id, story_id: id });
    showToast('🔖 Saved');
  }
  renderSaved();
}

2. Content Security Policy (CSP)

A CSP header tells browsers which external resources are allowed to load. Without one, a single XSS vulnerability can inject scripts from any domain. Add this to your Vercel config:

// vercel.json
{
  "headers": [
    {
      "source": "/(.*)",
      "headers": [
        {
          "key": "Content-Security-Policy",
          "value": "default-src 'self'; script-src 'self' 'unsafe-inline' https://cdn.jsdelivr.net https://cdnjs.cloudflare.com; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src https://fonts.gstatic.com; img-src 'self' data: https:; connect-src 'self' https://*.supabase.co https://api.polygon.io https://newsapi.org; frame-ancestors 'none';"
        },
        {
          "key": "X-Frame-Options",
          "value": "DENY"
        },
        {
          "key": "X-Content-Type-Options",
          "value": "nosniff"
        },
        {
          "key": "Referrer-Policy",
          "value": "strict-origin-when-cross-origin"
        },
        {
          "key": "Permissions-Policy",
          "value": "camera=(), microphone=(), geolocation=()"
        },
        {
          "key": "Strict-Transport-Security",
          "value": "max-age=63072000; includeSubDomains; preload"
        }
      ]
    }
  ]
}
Tip: Use securityheaders.com to scan your live site after deploying. Aim for an A or A+ rating.
3. XSS — Cross-Site Scripting

XSS is the most common web vulnerability. It happens when user input is inserted into the page as raw HTML, allowing attackers to inject scripts that steal session tokens, redirect users, or mine crypto.

The current code has an XSS risk in the community tab. When a user posts a comment and it's rendered with innerHTML, any HTML tags in the comment text will execute.

// VULNERABLE — never do this with user input
el.innerHTML = userComment;

// SAFE — escape all user-generated text before rendering
function escapeHtml(str) {
  return str
    .replace(/&/g, '&')
    .replace(//g, '>')
    .replace(/"/g, '"')
    .replace(/'/g, ''');
}

// Use in comment render:
el.innerHTML = `<div class="ctext">${escapeHtml(comment.text)}</div>`;

The rule is simple: never use innerHTML with user-provided text. Use textContent for plain text, or escape all HTML before injecting.

4. Rate Limiting

Without rate limiting, a single person can submit 10,000 comments, exhaust your API quota in minutes, or hammer your serverless functions with requests.

🚦 Vercel / Cloudflare WAF
Add rate limiting rules at the edge. Cloudflare free tier includes basic rate limiting — 10 rules, protects against floods.
Cost: Free on Cloudflare
🔁 Upstash Redis
Serverless Redis for token-bucket rate limiting in your API functions. Fastest option for per-user limits.
Cost: Free up to 10,000 req/day

Basic rate limiting in a Vercel serverless function using Upstash:

import { Ratelimit } from "@upstash/ratelimit";
import { Redis } from "@upstash/redis";

const ratelimit = new Ratelimit({
  redis: Redis.fromEnv(),
  limiter: Ratelimit.slidingWindow(10, "1 m"), // 10 requests per minute per IP
});

export default async function handler(req, res) {
  const ip = req.headers['x-forwarded-for'] || 'unknown';
  const { success } = await ratelimit.limit(ip);
  if (!success) return res.status(429).json({ error: 'Too many requests' });
  // ... rest of handler
}
5. Input Validation & Sanitization

Every piece of data that comes from a user must be validated before it touches your database or gets rendered on screen.

// Server-side validation in your API route (never trust frontend-only validation)
function validateComment(text) {
  if (typeof text !== 'string') return false;
  if (text.trim().length < 1) return false;
  if (text.length > 1000) return false;
  if (/[<>]/.test(text)) return false;  // Reject raw HTML tags
  return true;
}

function validateEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email) && email.length < 255;
}

function validateUrl(url) {
  try {
    const u = new URL(url);
    return ['http:', 'https:'].includes(u.protocol);
  } catch { return false; }
}
Never rely on frontend validation alone. Anyone can bypass your HTML form — they'll call your API directly with raw HTTP requests. All validation must be re-run server-side.
6. Stripe Payment Security

Stripe is secure by design — but only if you implement it correctly. The correct flow for your tool submission payments:

1
User fills out the Submit Tool form and clicks "Continue to Payment" — this calls your backend (not Stripe directly)
2
Your backend (Vercel function) validates the form data, then calls stripe.checkout.sessions.create() using your secret key
3
Stripe returns a session.url — your backend sends this URL to the frontend and the user is redirected to Stripe's hosted checkout page
4
After payment, Stripe calls your webhook endpoint (a separate serverless function) to confirm payment succeeded
5
Your webhook verifies the Stripe signature (to confirm it's genuinely from Stripe), then writes the approved submission to your database
// api/create-checkout.js — Vercel function
import Stripe from 'stripe';
const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

export default async function handler(req, res) {
  const { toolName, email, plan } = req.body;

  // Validate inputs server-side
  if (!toolName || !validateEmail(email)) {
    return res.status(400).json({ error: 'Invalid input' });
  }

  const session = await stripe.checkout.sessions.create({
    payment_method_types: ['card'],
    mode: plan === 'featured' ? 'subscription' : 'payment',
    line_items: [{
      price: plan === 'featured'
        ? process.env.STRIPE_FEATURED_PRICE_ID   // $149/mo recurring
        : process.env.STRIPE_STANDARD_PRICE_ID,  // $49 one-time
      quantity: 1,
    }],
    customer_email: email,
    metadata: { toolName, plan },
    success_url: `${process.env.SITE_URL}/success?session_id={CHECKOUT_SESSION_ID}`,
    cancel_url: `${process.env.SITE_URL}/`,
  });

  res.json({ url: session.url });
}

// api/stripe-webhook.js — receives Stripe events
export default async function handler(req, res) {
  const sig = req.headers['stripe-signature'];
  let event;
  try {
    event = stripe.webhooks.constructEvent(
      req.body, sig, process.env.STRIPE_WEBHOOK_SECRET
    );
  } catch (err) {
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }

  if (event.type === 'checkout.session.completed') {
    const session = event.data.object;
    // Now safe to write to database — payment confirmed
    await supabase.from('tool_submissions').insert({
      name: session.metadata.toolName,
      stripe_session_id: session.id,
      status: 'approved',
    });
  }
  res.json({ received: true });
}

8. CORS Policy

Cross-Origin Resource Sharing controls which domains can call your APIs. By default, your Vercel serverless functions should only accept requests from your own domain.

// Add to every API handler
export default async function handler(req, res) {
  // Only allow your own domain
  const allowed = ['https://actualintelligence.com', 'http://localhost:3000'];
  const origin = req.headers.origin;

  if (allowed.includes(origin)) {
    res.setHeader('Access-Control-Allow-Origin', origin);
  }
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization');

  if (req.method === 'OPTIONS') return res.status(200).end();
  // ... rest of handler
}

You currently have zero npm dependencies — this is actually a security advantage for the single-file version. When you move to Next.js, follow these rules:
